Get Started in 3 Steps

From zero to autonomous browsing in minutes.

1

Install

pip install openbrowser-ai

2

Write a Task

from openbrowser import CodeAgent, ChatGoogle

from dotenv import load_dotenv

import asyncio

load_dotenv()

async def main():

agent = CodeAgent(

task="Find the #1 post on Show HN",

llm=ChatGoogle(),

)

await agent.run()

asyncio.run(main())
